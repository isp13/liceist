using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class playerControl : MonoBehaviour {
	public bool Ranim = false;
	public bool Lanim = false;
	public bool UPanim = false;
	public bool Danim = false;

	private Rigidbody2D rb2d;
	public float moveSpeed;
	Animator anim;

	// Use this for initialization
	void Start () {
		rb2d = GetComponent<Rigidbody2D> ();
		anim = GetComponent<Animator>();
		int up = Animator.StringToHash("up");

		int down = Animator.StringToHash("down");

		int left = Animator.StringToHash("left");

		int right = Animator.StringToHash("right");

	}

	void isAlreadyPlaying()
	{
		
	}
	// Update is called once per frame
	void Update () {
		

		if(Input.GetAxisRaw ("Horizontal") > 0.5f){
			
			transform.Translate(new Vector3(Input.GetAxisRaw ("Horizontal") * moveSpeed * Time.deltaTime, 0f, 0f));

			anim.Play ("Right");
		}
		else if(Input.GetAxisRaw ("Horizontal") < -0.5f){
			
			transform.Translate(new Vector3(Input.GetAxisRaw ("Horizontal") * moveSpeed * Time.deltaTime, 0f, 0f));

			anim.Play ("Left");
		}

		else if(Input.GetAxisRaw ("Vertical") > 0.5f){
			
			transform.Translate(new Vector3(0f, Input.GetAxisRaw ("Vertical") * moveSpeed * Time.deltaTime, 0f));

			anim.Play ("Up");
		}

		else if(Input.GetAxisRaw ("Vertical") < -0.5f){
			
			transform.Translate(new Vector3(0f, Input.GetAxisRaw ("Vertical") * moveSpeed * Time.deltaTime, 0f));

			anim.Play ("Down");
		}

		//anim.SetFloat ("MoveX", Input.GetAxisRaw("Horizontal"));
		//anim.SetFloat ("MoveY", Input.GetAxisRaw("Vertical"));
	}
	}
